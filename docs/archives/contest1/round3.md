Justin wants to become the next crabber. To do so, he must kill scuttle crabs.

centered on the kill, causing them to run away. Given a 2D array, where 1 represents
the location of a scuttle crab, and 0 represents an empty spot, calculate the
maximum number of scuttle crabs Justin can kill.

number of columns, which are the dimensions of the 2D array.
The next 100 lines each have 150 space separated integers, either 0 or 1.

**Output Specifications**

        if (crabs[i][j] == 1) {
          int tmp = search(max(0, i - 2), min(row - 1, i + 2), max(0, j - 2), min(col - 1, j + 2), crabs);
          if (tmp < bs) {
            best_scuttle.first = i;
            best_scuttle.second = j;
            bs = tmp;
          }
        }
      }
    }

    ++res;
    scuttles -= bs;

    for (int i = max(0, best_scuttle.first - 2); i <= min(row - 1, best_scuttle.first + 2); ++i) {
      for (int j = max(0, best_scuttle.second - 2); j <= min(col - 1, best_scuttle.second + 2); ++ j) {
        crabs[i][j] = 0;
      }
    }
  }
  cout << res << endl;
}

// Use a bitset and some math to solve this problem

#include <bits/stdc++.h>

using namespace std;

int main() {
  vector<bitset<5000>> grid;

  ifstream infile;
  infile.open("input.txt");

  // We use a bitset (or a vector of bools) otherwise it will be too slow
  for (int i = 0; i < 5000; ++i) {
    string n;
    infile >> n;

    bitset<5000> b(n);
    grid.push_back(b);
  }

  infile.close();

  long long total = 0;
  for (int j = 0; j < 4999; ++j) {
    for (int k = j + 1; k < 5000; ++k) {
      // We count how many 1s are in the same column on both rows then calculate how many rectangles we can form and add it to the total
      long long count = (grid[j]&grid[k]).count();
      total = (total + count * (count - 1) / 2) % 100000000;
    }
  }
  cout << total << endl;
}

## Problem 3: Mountain Jumping

**Problem Statement**

Alice is very skilled at jumping. She is jumping across mountains. Using the height of the mountain to her advantage, she can jump $k$ units, where $k$ is the height of the mountain. It takes 1 second for each jump, regardless of distance. For any mountain that is of equal height or less than the one she is standing on, she can jump over them or directly onto their peaks. However, for mountains that have a greater height than the one she is standing on, she cannot jump over these mountains, and must spend an additional second climbing to the peak.

Given $n$ mountains and their respective heights, find the minimum amount of time in seconds needed to reach the last mountain peak if she starts on the peak of the first mountain.

**Input Specifications**
The first line has one integer, $n = 1000000$, the number of mountains.

The next line has $n$ space separated integers, $k_1, k_2, k_3, \cdots, k_n$, the heights of the mountains, with $k_i$ being the height of the $i$th mountain from the left for $1 \leq i \leq n$.

**Output Specifications**
Output a single integer, the time in seconds it takes for Alice to reach the last mountain peak. It is guaranteed that it will take less than $2^{31} - 1$ seconds.

**Example:** If there are 20 mountains and the heights are as following,

3, 1, 2, 4, 1, 5, 3, 1, 1, 2, 1, 3, 8, 1, 1, 4, 3, 9, 3, 1

then it will take 12 seconds.

// In order to jump as few times as possible, we want to choose the mountain that is both tall and far from the start as the height of the mountain is equivalent to the jump distance
// we sum the height and its indice; this is the value of the mountain each time
// we choose the earliest and most valuable mountain in range

#include <bits/stdc++.h>

using namespace std;

int main() {
    int n;
    
    // Mountains
    vector<int> nums;
    // Values of mountains
    vector<int> nums2;

    ifstream infile;
    infile.open("input.txt");

    infile >> n;

    for (int i = 0; i < n; ++i) {
        int temp;
        infile >> temp;
        nums.push_back(temp);
    }

    infile.close();

    for (int i = 0; i < n - 1; ++i) {
        nums2.push_back(i);
        // The value of the mountains is limited by the next tallest mountain
        for (int j = i + 1; j <= min(i + nums[i], n - 1); ++j) {
            ++nums2[i];
            if (nums[j] > nums[i]) break;
        }
    }
    nums2.push_back(INT_MAX);

    int jumps = 0;
    int step = 0;
    
    while (step < n - 1) {
        int max = step;

        for (int i = step + 1; i <= min(step + nums[step], n - 1); ++i) {
            // Looking for highest valued mountain
            if (nums2[i] > nums2[max]) {
                max = i;
                // If the mountain is taller, we stop searching as we cannot jump over it also note we only need to check here as the value of a taller mountain in range is always the greatest. Can you prove this?
                if (nums[max] > nums[step]) {
                    ++jumps;
                    break;
                }
            }
        }

        step = max;
        ++jumps;
    }
    
    cout << jumps << endl;
}
